022    package com.skype.ipc;
023    
024    import java.io.EOFException;
025    import java.io.IOException;
026    import java.util.Collections;
027    import java.util.HashMap;
028    import java.util.Map;
029    import java.util.concurrent.BlockingQueue;
030    import java.util.concurrent.LinkedBlockingQueue;
031    
032    import com.skype.ipc.RootObject.ErrorListener;
033    import com.skype.util.Log;
034    
035    public class ResponseListener extends Thread
036    {
037        private static final String                      TAG                    = "ResponseListener";
038        private static final int                         XCALL_RESPONSE_TIMEOUT = 60000;
039        private Thread                                   my_thread;
040        private Transport                                ioTransport;
041        private boolean                                  die;
042        private final Map<Integer, Response>             response_table;
043        private final BlockingQueue<GetPropertyResponse> get_prop_response_queue;
044        private EventDispatcher                          myEdp;
045    
046        private boolean                                  responseWaitInterrupted;
047        private ErrorListener                            errorListener;
048    
049        public ResponseListener(EventDispatcher edp, RootObject root)
050        {
051            myEdp = edp;
052            ioTransport = root.getTransport();
053            errorListener = root.errorListener;
054            setName("response listener thread");
055            die = false;
056            response_table = Collections.synchronizedMap(new HashMap<Integer, Response>());
057            get_prop_response_queue = new LinkedBlockingQueue<GetPropertyResponse>();
058            responseWaitInterrupted = false;
059        }
060    
061        public boolean isDead()
062        {
063            return die;
064        }
065        
066        public void die()
067        {
068            die = true;
069            my_thread.interrupt(); // trigger interrupt in case of waiting
070        }
071    
072        synchronized private byte readByte() throws IOException
073        {
074            int res = ioTransport.read();
075            byte resByte = (byte) (res & 0xFF);
076            Log.d(TAG, "ResponseListener::readByte: " + Integer.toHexString(res) + " ('" + (char) res + "') " + resByte);
077    
078            return resByte;
079        }
080    
081        public void run()
082        {
083            my_thread = Thread.currentThread();
084            Log.d(TAG, "Running! Threadid=" + my_thread.getId());
085    
086            try {
087                byte b1 = readByte();
088                byte b2 = readByte();
089                if (b1 == 'O' && b2 == 'K') {
090                    Log.d(TAG, "Skypekit executable handshake successful");
091                }
092                else if (b1 == 'E' && b2 == 'R') {
093                    // service unavailable
094                    Log.e(TAG, "Skypekit executable handshake failed. Code '" + (char) b1 + (char) b2 + "'.");
095                }
096                else if ((b1 == 'A' && b2 == 'N') ||
097                        (b1 == 'O' && b2 == 'B') ||
098                        (b1 == 'S' && b2 == 'N') ||
099                        (b1 == 'O' && b2 == 'M') ||
100                        (b1 == 'A' && b2 == 'B') ||
101                        (b1 == 'S' && b2 == 'B') ||
102                        (b1 == 'P' && b2 == 'B'))
103                {
104                    // application token rejected for one reason or another
105                    Log.e(TAG, "Skypekit executable handshake failed. Code '" + (char) b1 + (char) b2 + "'.");
106                }
107                else {
108                    // unknown error in handshake
109                    Log.e(TAG, "Unknown Skypekit executable handshake failure. Code '" + (char) b1 + (char) b2 + "'.");
110                }
111    
112            }
113            catch (IOException e) {
114                Log.e(TAG, "exception=", e);
115                e.printStackTrace();
116            }
117            catch (Exception e) {
118                Log.e(TAG, "Unknown exception caught = ", e);
119                e.printStackTrace();
120                die = true;
121            }
122    
123            while ( ! die) {
124                try {
125                    int prefix = readByte();
126                    Log.d(TAG, "Prefix read:" + prefix);
127                    
128                    if (prefix == 'Z') {
129                        int cmd = readByte();
130                        Log.d(TAG, "Processing command " + cmd + " (" + (char) cmd + ")");
131    
132                        switch (cmd) {
133                        case 'r':
134                            x_call_response();
135                            break;
136                        case 'C':
137                            decode_property_change();
138                            break;
139                        case 'E':
140                            decode_event();
141                            break;
142                        case 'g':
143                            decode_get_response();
144                            break;
145                        default:
146                            Log.e(TAG, "Unknown command " + cmd + " (" + (char) cmd + ")");
147                        }
148                    }
149                    else {
150                        Log.e(TAG, "Bad symbol in the stream: " + prefix + " (" + (char) prefix + ") - discarding");
151                    }
152                }
153                catch (java.net.SocketException e) {
154                    Log.d(TAG, "Socket exception after normal socket shutdown");
155                }
156                catch (EOFException e) {
157                    Log.e(TAG, "EOF Exception, no socket connection:" + e.getMessage());
158                    if (errorListener != null)
159                        errorListener.OnSkypeKitConnectionClosed();
160                    die = true;
161                }
162                catch (IOException e) {
163                    Log.e(TAG, "Listener IO Exception:" + e.getMessage());
164                    e.printStackTrace();
165                    die = true;
166                }
167                catch (Exception e) {
168                    Log.e(TAG, "Unknown exception caught in Listener:", e);
169                    e.printStackTrace();
170                    die = true;
171                }
172            }
173            Log.d(TAG, "Exiting!");
174        }
175    
176        void x_call_response() throws IOException
177        {
178            int rid = AbstractDecoder.decodeOneVaruint(ioTransport);
179            Log.d(TAG, "x_call_response rid: " + rid);
180    
181            // Tell SkypeKit that we got new response in case it's waiting for it
182            synchronized (response_table) {
183                response_table.put(rid, new Response(ioTransport, rid));
184    
185                response_table.notifyAll();
186                Log.d(TAG, "Notified of incoming response rid=" + rid);
187            }
188            ;
189        }
190    
191        void decode_property_change() throws IOException
192        {
193            Log.d(TAG, "->Property change - decode_property_change starting...");
194            try {
195                myEdp.AddPropertyChange(new PropertyChange(ioTransport));
196            }
197            catch (InterruptedException e) {
198                throw new IOException("AddPropertyChange interrupted at EventDispatcher");
199            }
200            Log.d(TAG, "->decode_property_change done.");
201        }
202    
203        void decode_event() throws IOException
204        {
205            Log.d(TAG, "->Event - decode_event starting...");
206            try {
207                myEdp.AddEvent(new Event(ioTransport));
208            }
209            catch (InterruptedException e) {
210                throw new IOException("AddEvent interrupted at EventDispatcher");
211            }
212            Log.d(TAG, "->decode_event done.");
213        }
214    
215        void decode_get_response() throws IOException
216        {
217            // SkypeKit guarantees that get responses always arrive in same order
218            // they were requested
219            Log.d(TAG, "->Event - decode_get_response starting...");
220            try {
221                if (!responseWaitInterrupted) {
222                    // place the response in the queue
223                    get_prop_response_queue.put(new GetPropertyResponse(ioTransport));
224                }
225                else {
226                    new GetPropertyResponse(ioTransport); // discard the response
227                    responseWaitInterrupted = false;
228                }
229            }
230            catch (InterruptedException e) {
231                throw new IOException("Interrupted while get_prop_response_queue.put(..).");
232            }
233            Log.d(TAG, "->Decoding done, placed GetResponse to queue");
234        }
235    
236        // This is called by SkypeKit thread
237        // NOTE: That sending property request and calling get_prop_response() to get
238        // the response MUST be synchronized
239        public GetPropertyResponse get_prop_response()
240        {
241            try {
242                Log.d(TAG, "Going to take GetResponse from queue");
243                GetPropertyResponse temp = get_prop_response_queue.take();
244                if (temp.isMultiresponse()) {
245                    Log.d(TAG, "Took GetResponse from queue. It is a multiresponse.");
246                }
247                else {
248                    Log.d(TAG, "Took GetResponse from queue. Propid=" + temp.getPropId());
249                }
250                return temp;
251            }
252            catch (InterruptedException e) {
253                responseWaitInterrupted = true;
254                Log.d(TAG, "get_prop_response_queue.take() was interrupted! Next property response will be discarded to keep queue in sync.");
255                return null;
256            }
257        }
258    
259        // This is called by SkypeKit thread
260        public Response get_x_call_response(int rid) throws InterruptedException
261        {
262            Response myresponse;
263            while (!die) {
264                myresponse = response_table.remove(rid);
265                if (myresponse == null) {
266                    try {
267                        Log.d(TAG, "Response not ready for rid=" + rid + ". Going to wait.");
268                        synchronized (response_table) {
269                            if (!response_table.containsKey(rid))
270                                response_table.wait(XCALL_RESPONSE_TIMEOUT);
271                        }
272                        Log.d(TAG, "Woken up for rid=" + rid);
273                    }
274                    catch (InterruptedException e) {
275                        Log.e(TAG, "Waiting for rid=" + rid + " interrupted. e:" + e.toString());
276                        throw e;
277                    }
278                }
279                else {
280                    // response has arrived
281                    Log.d(TAG, "Response has arrived for rid=" + rid + ". Processing...");
282                    return (myresponse);
283                }
284            }
285            Log.d(TAG, "Exiting GetXCallResponse due to Response Listener being killed.");
286            throw new InterruptedException("Response listener killed.");
287        }
288    }
