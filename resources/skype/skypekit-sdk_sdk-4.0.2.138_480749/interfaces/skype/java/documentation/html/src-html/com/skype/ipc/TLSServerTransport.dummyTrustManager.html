117            void do_handshake() throws SSLException, IOException 
118            {
119                    
120                    Log.d(TAG, "do_handshake()");
121                    SSLEngineResult engineResult = null;
122                    HandshakeStatus result =  HandshakeStatus.NEED_UNWRAP;
123                    while(true) {
124                        switch(result) {
125                        case FINISHED:
126                            Log.d(TAG, result.toString());
127                            return;
128                        case NEED_TASK:
129                            Log.d(TAG, result.toString());
130                            Runnable task;
131                            while((task=engine.getDelegatedTask()) != null) {
132                                task.run();
133                            }
134                            break;
135                        case NEED_UNWRAP: {
136    
137                            byte [] msg = new byte [16384];
138                            int idx = 0;
139                            Log.d(TAG, result.toString());
140    
141                            if (decrypted_bytes.length == 0) {
142                                try {
143                                    transport.peek();
144                                    while(transport.hasMore() ) {
145                                        msg[idx++] = (byte) transport.read();
146                                    }
147                                }
148                                catch (IOException e) {
149                                    Log.e(TAG, "Read exception:" + e.getMessage() + ":" + e.getCause());
150                                    throw e;
151                                }
152                            }
153                            else {
154                                idx = decrypted_bytes.length;
155                                System.arraycopy(decrypted_bytes, 0, msg, 0, idx);
156                                decrypted_bytes = new byte[]{};
157                            }
158    
159                            ByteBuffer src0 = ByteBuffer.wrap(msg);
160                            ByteBuffer dst0 = ByteBuffer.allocate(16384);
161                            engineResult = engine.unwrap(src0, dst0);
162                            result = engineResult.getHandshakeStatus();
163                            int bytesConsumed = engineResult.bytesConsumed();
164    
165                            int sum = 0; //check if we had trailing zeroes
166                            while(bytesConsumed < idx)
167                                sum |= msg[bytesConsumed++];
168    
169                            int bytesLeftOver = idx - engineResult.bytesConsumed();
170                            if (sum != 0 && bytesLeftOver > 0 ) { //more than one message, keep a copy
171                                decrypted_bytes = new byte[bytesLeftOver];
172                                System.arraycopy(msg, engineResult.bytesConsumed(), 
173                                        decrypted_bytes, 0, bytesLeftOver);
174                            }
175                            break;
176                        }
177                        case NEED_WRAP: {
178                            Log.d(TAG, result.toString());
179                            ByteBuffer EMPTY = ByteBuffer.allocate(0);
180                            ByteBuffer dst1 = ByteBuffer.allocate(65536);
181                            engineResult = engine.wrap(EMPTY, dst1);
182                            byte [] bytes2send = new byte[engineResult.bytesProduced()];
183                            dst1.rewind();
184                            dst1.get(bytes2send);
185                            transport.write(bytes2send.length , bytes2send);
186                            break;
187                        }
188                        case NOT_HANDSHAKING:
189                            Log.d(TAG, "discarding " + decrypted_bytes.length + " bytes");
190                            if (decrypted_bytes.length > 0) {
191                                ByteBuffer s = ByteBuffer.wrap(decrypted_bytes);
192                                Log.hexdump(s.array());
193                                decrypted_bytes = new byte[] {};
194                            }
195                            Log.d(TAG, result.toString());
196                            return;
197                        }
198                        result = engine.getHandshakeStatus();
199    
200            }

209            @Override
210            public boolean connect() throws IOException
211            {
212                    Log.d(TAG, "connect()");
213    
214                    if (!transport.connect()) 
215                            return false;
216    
217                    try {
218                            context = SSLContext.getInstance("TLS");
219                    } catch (NoSuchAlgorithmException e) {
220                            throw new IOException("TLSServerTransport:IOException:" + e.getMessage() );
221                    }
222                    KeyManager [] km = { keyManager };
223                    TrustManager [] tm = { trustManager };
224                    try {
225                            context.init(km, tm, null);
226                    } catch (KeyManagementException e) {
227                            throw new IOException("TLSServerTransport:KeyManagementException:" + e.getMessage() );
228                    }
229                    engine = context.createSSLEngine();
230                    engine.setUseClientMode(false);
231                    engine.setNeedClientAuth(false);
232                    engine.setWantClientAuth(false);
233    
234                    try {
235                            @SuppressWarnings("rawtypes")
236                Class c = Class.forName("javax.net.ssl.SSLEngine");
237                            for (Method m : c.getDeclaredMethods() ) {
238                                    if (m.getName() == "getSSLParameters") { //this is not supported on Android < 2.3
239                                            SSLParameters params = engine.getSSLParameters();
240                                            params.setCipherSuites(new String [] { 
241                                                    "TLS_RSA_WITH_AES_128_CBC_SHA"
242                                            });
243                                            engine.setSSLParameters(params);
244                                            }
245                                    }
246                            }
247                    catch (ClassNotFoundException e) { //shouldnt ever happen
248                    }
249    
250                    try {
251                            do_handshake();
252                    } 
253                    catch (SSLException e) {
254                        throw new IOException("TLSServerTransport:handshake_failed:" + e.getMessage(), e);
255                    }
256                    catch (IOException e) {
257                        throw new IOException("TLSServerTransport:handshake_failed:" + e.getMessage(), e);
258                    }
259                    connected = true;
260    
261                    X509Certificate cert = keyManager.getCertificateChain("dummyServer")[0];
262                    byte[] certBytes = null;
263                    try {
264                            certBytes = cert.getEncoded();
265                    } catch (CertificateEncodingException e) {
266                            e.printStackTrace();
267                    }
268                    String lenStr = String.format("%08X", certBytes.length);
269                    byte [] lenBytes = lenStr.getBytes();
270                    write(lenBytes.length,lenBytes);
271                    write(certBytes.length,certBytes);
272    
273                    return true;
274            }

299            @Override
300            public int peek() throws IOException
301            {
302    //          Log.d(TAG, "TLSServerTransport::peek() called");
303                checkConnected();
304    
305                if (decrypted_chunks.isEmpty()) {
306                    readChunks();
307                }
308                if (decrypted_chunks.isEmpty())
309                {
310                    Log.d(TAG, "TLS Peek - nothing to peek at...");
311                    return 0;
312                }
313    //          Log.d(TAG, "TLS Peek: " + Integer.toHexString(decrypted_chunks.getFirst()[0] & 0xFF) +
314    //                  " " + (char)decrypted_chunks.getFirst()[0] +
315    //                  " " + (int)(decrypted_chunks.getFirst()[0] & 0xFF));
316                return decrypted_chunks.getFirst()[0];
317            }

329            @Override
330            public int read(int numBytes, byte[] bytes) throws IOException
331            {
332                    checkConnected();
333    
334                    int num_waits = 0;
335                    final int MAXWAITS = 10;
336                    final int WAITMILIS = 50; // how long should we wait?
337                    
338                    while (decrypted_chunks.isEmpty())
339                    {
340                        readChunks();
341    
342                        if (decrypted_chunks.isEmpty()) {
343                            if (++num_waits > MAXWAITS)
344                                throw new IOException("timeout waiting for transport input");
345    
346                            try {
347                                Log.d(TAG, "Waiting for data on transport take:" + num_waits);
348                                wait(WAITMILIS);
349                            }
350                            catch (InterruptedException e) {}
351                        }
352                    }
353    
354                    byte [] chunk = decrypted_chunks.removeFirst();
355    //              Log.d(TAG, "chunk being read from is:");
356    //              Log.hexdump(chunk);
357    
358                    if (chunk.length == numBytes) { 
359                        // copy decrypted bytes into caller's byte array
360                        System.arraycopy(chunk, 0, bytes, 0, chunk.length);
361                        return chunk.length;
362                    }
363                    else if (chunk.length < numBytes) {
364                        throw new IOException("Input stream parsing error. underflow?");
365                    }
366                    else {
367                        // unfortunately, our readers do not request the input in chunks
368                        // instead they read 1 byte increments which means we have to pop
369                        // the top chunk of input, peel one byte off, and push the remains back onto the list.
370                        // sub-optimal but works for now.
371                        System.arraycopy(chunk, 0, bytes, 0, numBytes);
372                        byte [] chunkless = new byte[chunk.length - numBytes];
373                        System.arraycopy(chunk, numBytes, chunkless, 0, chunk.length-numBytes);
374                        decrypted_chunks.addFirst(chunkless);
375                        return numBytes;
376                    }
377            }

380            public int read(int numBytes, byte[] bytes, boolean needNumBytes) throws IOException {
381                    Log.d(TAG, "TLSServerTransport::read(3) called");
382                    int bytesRead = read(numBytes, bytes);
383                    if ((bytesRead != numBytes) && (needNumBytes))
384                            throw new IOException("TLSServerTransport::read() - read less than required number of bytes");
385                    return bytesRead;

388            /**
389             * drain all input from transport stream and decrypt using engine.
390             * 
391             * @param raw_bytes - buffer to accept unwrapped bytes read from socket
392             * @throws IOException - when read buffer is too small, or other socket read exceptions thrown
393             */
394            protected void readChunks() throws IOException
395            {
396    
397                    boolean hadLeftovers = false;
398    
399                    // allocate byte buffers to match session parameters exactly
400                    if (raw_source == null) {
401                        SSLSession s = engine.getSession();
402                        MAXBUF = s.getPacketBufferSize();
403                        Log.d(TAG, "SSLSession packet size:" + MAXBUF);
404                        raw_source = ByteBuffer.allocate(MAXBUF);
405                    }
406    
407                    int pos = raw_source.position();
408                    if (pos > 0) {
409                        Log.d(TAG, "UNDERFLOW -contining from position:" + pos);
410                        hadLeftovers = true;
411                    }
412    
413                    transport.peek();
414                    while( transport.hasMore() &&
415                            raw_source.position() < MAXBUF ) {
416                        raw_source.put((byte)transport.read());
417                    }
418    
419                    if (pos == raw_source.position()) {
420                        Log.d(TAG, "Nothing read.");
421                        return;
422                    }
423                    Log.d(TAG, "readChunks() done reading at position:" + raw_source.position());
424    
425                    raw_source.flip();
426    
427                    ByteBuffer dst = ByteBuffer.allocate(MAXBUF); 
428                    boolean unwrapped;
429                    do {
430                        if (hadLeftovers) {
431                            dumpSourceBuffer("Before unwrap()");
432                        }
433    
434                        SSLEngineResult engineResult = engine.unwrap(raw_source, dst);
435    
436                        if (hadLeftovers) {
437                            dumpSourceBuffer("After  unwrap()");
438                        }
439    
440                        SSLEngineResult.Status unwrap_status = engineResult.getStatus();
441                        Log.d(TAG, "unwrap status:" + unwrap_status);
442    
443                        unwrapped = (unwrap_status == SSLEngineResult.Status.OK);
444                        dst.rewind();
445                        if (unwrapped)
446                        {
447                            int l = engineResult.bytesProduced();
448                            Log.d(TAG, "decrypted length:" + l);
449                            byte [] chunk = new byte[l];
450                            dst.get(chunk, 0, l);
451    
452                            Log.d(TAG, "\nchunk read is :");
453                            Log.hexdump(chunk);
454                            if (isLogging)
455                                logFileIn.write(chunk);
456    
457                            decrypted_chunks.addLast(chunk);
458                        }
459                        else if (unwrap_status != Status.BUFFER_UNDERFLOW) {
460                            throw new IOException("Unrecoverable TLS error: " + unwrap_status);
461                        }
462                        else {
463                            Log.d(TAG, "UNDERFLOW? reseting raw_source for next time.");
464                        }
465                        dst.rewind();
466                    } while (unwrapped && raw_source.hasRemaining());
467    
468                Log.d(TAG, "readChunks() done, hasRemaining:" + raw_source.hasRemaining());
469    
470                if (raw_source.hasRemaining()) {
471                    Log.e(TAG, "UNDERFLOW? bytes left over");
472                    raw_source.compact();
473                }
474                else {
475                    raw_source.clear();
476                }
477            }

479            private void dumpSourceBuffer(String tag)
480            {
481                int pos = raw_source.position();
482                int lim =raw_source.limit();
483            Log.e(TAG, "---" + tag + " position:" + pos + " limit:" +  lim);
484            
485            if (raw_source.hasArray() && pos < lim) {
486                byte [] src = raw_source.array();
487                byte [] section = new byte[lim - pos];
488                System.arraycopy(src, pos, section, 0, lim - pos);
489                Log.hexdump(section);
490            }
491            }
