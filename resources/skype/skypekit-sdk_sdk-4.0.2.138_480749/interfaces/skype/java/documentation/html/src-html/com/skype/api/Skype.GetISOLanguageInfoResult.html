099            public void HandlePropertyChange(PropertyChange pc){
100                    SkypeObject obj = null;

104                    //check object exists in list and get value 
105                    if (object_list.containsKey(pc.oid))

109                            
110                            switch (kind) {

112                                    case 'u':

114                                            try {

160                                    obj.mPropCache.put(pc.propid, val);

162                            else {
163                                    obj.mPropCache.remove(pc.propid);

165                            //call listener if this exists?

167                             {
168                                    Object listener = listeners.get(pc.moid);
169                                    if (listener != null){
170                                            prop = obj.GetPropertyAsEnum(pc.propid);
171                                            
172                                            switch (pc.moid)

233            public void FlushObjects()
234            {
235                    Iterator<Integer> iter = object_list.keySet().iterator();
236                    while (iter.hasNext()) {
237                            Integer key = iter.next();
238                            SkypeObject obj  = object_list.get(key);
239                            obj.flush_cache();
240                    }
241                    object_list.clear();
